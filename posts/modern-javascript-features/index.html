Modern JavaScript Features

JavaScript has evolved significantly over the years, introducing powerful features that make development more efficient and code more readable. Let’s explore some of the most important modern JavaScript features.

// Array destructuring
const colors = ['red', 'green', 'blue'];
const [first, second, third] = colors;
console.log(first); // 'red'

// Object destructuring
const person = {
    name: 'John',
    age: 30,
    city: 'New York'
};

const { name, age, city } = person;
console.log(name); // 'John'

// With default values
const { country = 'USA' } = person;
console.log(country); // 'USA'

// Object spreading
const user = { name: 'John', age: 30 };
const updatedUser = { ...user, city: 'Boston' };
console.log(updatedUser); // { name: 'John', age: 30, city: 'Boston' }

// Function arguments
const maxNumber = Math.max(...numbers);
console.log(maxNumber); // 3

// Traditional promise chain
function fetchUserData(userId) {
    return fetch(`/api/users/${userId}`)
        .then(response => response.json())
        .then(data => console.log(data))
        .catch(error => console.error(error));
}

// With async/await
async function fetchUserData(userId) {
    try {
        const response = await fetch(`/api/users/${userId}`);
        const data = await response.json();
        console.log(data);
    } catch (error) {
        console.error(error);
    }
}

// Multiple async operations
async function fetchMultipleResources() {
    try {
        const [users, posts] = await Promise.all([
            fetch('/api/users').then(r => r.json()),
            fetch('/api/posts').then(r => r.json())
        ]);
        
        return { users, posts };
    } catch (error) {
        console.error('Failed to fetch resources:', error);
    }
}

// math.js - Exporting
export const PI = 3.14159;

export function add(a, b) {
    return a + b;
}

export default function multiply(a, b) {
    return a * b;
}

// main.js - Importing
import multiply, { PI, add } from './math.js';

console.log(PI); // 3.14159
console.log(add(2, 3)); // 5
console.log(multiply(4, 5)); // 20

// Importing everything
import * as math from './math.js';
console.log(math.PI); // 3.14159

// Traditional approach (problematic with falsy values)
const count = value || 0; // This would replace 0, '', false with 0

// With nullish coalescing
const count = value ?? 0; // Only replaces null/undefined with 0

// Practical example
const settings = {
    theme: 'dark',
    fontSize: 0, // User explicitly set to 0
    autoSave: false // User explicitly set to false
};

const theme = settings.theme ?? 'light'; // 'dark'
const fontSize = settings.fontSize ?? 14; // 0 (not replaced)
const autoSave = settings.autoSave ?? true; // false (not replaced)

class BankAccount {
    #balance = 0; // Private field
    
    constructor(initialBalance) {
        this.#balance = initialBalance;
    }
    
    deposit(amount) {
        if (amount > 0) {
            this.#balance += amount;
        }
    }
    
    getBalance() {
        return this.#balance;
    }
    
    #calculateInterest() { // Private method
        return this.#balance * 0.05;
    }
}

const account = new BankAccount(1000);
console.log(account.getBalance()); // 1000
console.log(account.#balance); // Error: Private field
